export default function App() {

      setEvents(snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() })));
    }, (err) => console.error("Events fetch error:", err));

    const teamsRef = collection(db, 'artifacts', appId, 'users', user.uid, 'teams');

      setTeams(snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() })));
    }, (err) => console.error("Teams fetch error:", err));
    return () => unsubscribe();
  }, [user, isAdmin]);

  // Firestore Sync - All Registrations & Scores (Public)

    if (!user) return;
    const allTeamsRef = collection(db, 'artifacts', appId, 'public', 'data', 'registrations');

      setAllRegistrations(snap.docs.map(doc => ({ id: doc.id, ...doc.data() })));
    });

      setScores(snap.docs.map(doc => ({ id: doc.id, ...doc.data() })));
    });
    
    return () => { unsubTeams(); unsubScores(); };
  }, [user]);

  if (loading) return <div className="min-h-screen bg-slate-50 flex items-center justify-center"><div className="animate-spin rounded-full h-12 w-12 border-b-2 border-blue-600"></div></div>;

  return (
    <div className="min-h-screen bg-slate-50 flex flex-col md:flex-row">
      <nav className="w-full md:w-64 bg-white border-r border-slate-200 p-4 flex flex-col">
        <div className="flex items-center gap-2 px-2 mb-8 mt-2">
          <div className="bg-blue-600 p-2 rounded-lg text-white"><Trophy size={20} /></div>
          <span className="font-bold text-slate-800 text-lg">RoboReg v1</span>
        </div>

        <div className="space-y-1 flex-1">
          <NavItem active={view === 'dashboard'} onClick={() => setView('dashboard')} icon={<Users size={18}/>} label="Dashboard" />
          {!isAdmin && <NavItem active={view === 'teams'} onClick={() => setView('teams')} icon={<School size={18}/>} label="My Teams" />}
          
          <div className="pt-4 pb-2 px-2 text-xs font-semibold text-slate-400 uppercase tracking-wider">Competition</div>
          <NavItem active={view === 'judging'} onClick={() => setView('judging')} icon={<ClipboardCheck size={18}/>} label="Scoring / Judging" />
          
          {isAdmin && (
            <>
              <div className="pt-4 pb-2 px-2 text-xs font-semibold text-slate-400 uppercase tracking-wider">Admin Controls</div>
              <NavItem active={view === 'admin-teams'} onClick={() => setView('admin-teams')} icon={<Shield size={18}/>} label="All Registrations" />
              <NavItem active={view === 'admin-events'} onClick={() => setView('admin-events')} icon={<Settings size={18}/>} label="Manage Events" />
            </>
          )}
        </div>

        <div className="pt-4 border-t border-slate-100 space-y-2">
          <button onClick={() => setIsAdmin(!isAdmin)} className={`flex items-center gap-3 w-full px-3 py-2 text-xs font-bold rounded-md transition-all border border-dashed ${isAdmin ? 'bg-amber-50 text-amber-700 border-amber-200' : 'text-slate-400 border-slate-200'}`}>
            {isAdmin ? <Lock size={14} /> : <Shield size={14} />}
            {isAdmin ? "Exit Admin Panel" : "Open Admin Panel"}

        </div>

      <main className="flex-1 overflow-y-auto p-4 md:p-8">
        <header className="mb-8 flex justify-between items-start">
          <div>
            <h1 className="text-2xl font-bold text-slate-900 capitalize">{view.replace('-', ' ')}</h1>
            <p className="text-slate-500 text-sm">{isAdmin ? 'Administrative Overview' : `Logged in as ${user.displayName || 'Coach'}`}</p>

          {isAdmin && view === 'admin-events' && <ShareJudgingLink />}

// --- View Components ---

function NavItem({ active, onClick, icon, label }) {
  return (
    <button onClick={onClick} className={`flex items-center gap-3 w-full px-3 py-2 text-sm font-medium rounded-md transition-all ${active ? 'bg-blue-50 text-blue-700 shadow-sm' : 'text-slate-600 hover:bg-slate-100 hover:text-slate-900'}`}>
      {icon} {label}
    </button>
  );
}

function ShareJudgingLink() {
  const [copied, setCopied] = useState(false);
  
  const copyLink = () => {
    const url = new URL(window.location.href);
    url.searchParams.set('view', 'judging');
    document.execCommand('copy'); // Fallback as per instructions
    const dummy = document.createElement("input");
    document.body.appendChild(dummy);
    dummy.value = url.toString();
    dummy.select();
    document.execCommand("copy");
    document.body.removeChild(dummy);
    setCopied(true);
    setTimeout(() => setCopied(false), 2000);
  };

  return (
    <button 
      onClick={copyLink}
      className={`flex items-center gap-2 px-4 py-2 rounded-lg text-sm font-bold transition-all shadow-sm ${copied ? 'bg-emerald-100 text-emerald-700' : 'bg-white border border-slate-200 text-slate-600 hover:bg-slate-50'}`}
    >
      {copied ? <CheckCircle size={16}/> : <Share2 size={16}/>}
      {copied ? 'Judging Link Copied!' : 'Copy Judging Link'}
    </button>
  );
}

// --- Scoring & Judging ---

function JudgingView({ events, registrations, scores }) {
  const [selectedEvent, setSelectedEvent] = useState(null);
  const [selectedTeam, setSelectedTeam] = useState(null);
  const [rawValue, setRawValue] = useState('');

  const activeEvent = events.find(e => e.id === selectedEvent);
  
  const submitScore = async () => {
    if (!selectedEvent || !selectedTeam) return;
    const scoreId = `${selectedEvent}_${selectedTeam}`;
    try {
      await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'scores', scoreId), {
        eventId: selectedEvent,
        teamId: selectedTeam,
        rawValue: parseFloat(rawValue),
        judgeId: auth.currentUser.uid,
        updatedAt: new Date().toISOString()
      });
      setRawValue('');
      setSelectedTeam(null);
    } catch (err) { console.error(err); }
  };

  const getExistingScore = (tid) => scores.find(s => s.eventId === selectedEvent && s.teamId === tid);

  return (
    <div className="space-y-6">
      <div className="bg-white p-6 rounded-xl border border-slate-200 shadow-sm">
        <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
          <div>
            <label className="block text-sm font-bold text-slate-700 mb-2">1. Select Competition Event</label>
            <select 
              className="w-full p-3 border border-slate-200 rounded-lg"
              value={selectedEvent || ''}
              onChange={e => setSelectedEvent(e.target.value)}
            >
              <option value="">Choose an event...</option>
              {events.map(e => <option key={e.id} value={e.id}>{e.name}</option>)}
            </select>
            {activeEvent && (
              <div className="mt-3 p-3 bg-blue-50 border-l-4 border-blue-400 text-xs text-blue-800">
                <strong>Method:</strong> {activeEvent.method === 'relative' ? 'Proximity-to-Leader Tiering' : 'Absolute Scoring'}<br/>
                <strong>Goal:</strong> {activeEvent.type.includes('fast') ? 'Lower is better' : 'Higher is better'}
              </div>
            )}
          </div>

          <div>
            <label className="block text-sm font-bold text-slate-700 mb-2">2. Select Team</label>
            <select 
              className="w-full p-3 border border-slate-200 rounded-lg"
              disabled={!selectedEvent}
              value={selectedTeam || ''}
              onChange={e => setSelectedTeam(e.target.value)}
            >
              <option value="">Choose a team...</option>
              {registrations.map(r => (
                <option key={r.id} value={r.id}>
                  {r.teamName} ({r.schoolName}) {getExistingScore(r.id) ? '✓' : ''}
                </option>
              ))}
            </select>
          </div>
        </div>

        {selectedTeam && (
          <div className="mt-8 pt-8 border-t border-slate-100 animate-in fade-in slide-in-from-bottom-4">
            <h4 className="font-bold text-slate-800 mb-4 flex items-center gap-2">
              <ClipboardCheck className="text-blue-600" />
              Enter Data for {registrations.find(r => r.id === selectedTeam)?.teamName}
            </h4>
            <div className="flex gap-4">
              <input 
                type="number"
                step="0.01"
                className="flex-1 p-4 border-2 border-slate-200 rounded-xl text-xl font-bold focus:border-blue-500 outline-none"
                placeholder={activeEvent?.type.includes('time') ? "Seconds (e.g. 45.2)" : "Score Value"}
                value={rawValue}
                onChange={e => setRawValue(e.target.value)}
              />
              <button 
                onClick={submitScore}
                className="bg-blue-600 text-white px-8 py-4 rounded-xl font-bold hover:bg-blue-700 transition-all shadow-lg shadow-blue-100"
              >
                Submit Result
              </button>
            </div>
          </div>
        )}
      </div>

      {selectedEvent && (
        <div className="bg-slate-900 rounded-xl p-6 text-white shadow-xl">
          <h4 className="flex items-center gap-2 font-bold mb-4">
            <Award className="text-amber-400" /> Current Raw Standings
          </h4>
          <div className="space-y-2">
            {scores
              .filter(s => s.eventId === selectedEvent)
              .sort((a,b) => activeEvent?.type.includes('fast') ? a.rawValue - b.rawValue : b.rawValue - a.rawValue)
              .map((s, i) => {
                const team = registrations.find(r => r.id === s.teamId);
                return (
                  <div key={s.id} className="flex justify-between items-center bg-white/5 p-3 rounded-lg border border-white/10">
                    <div className="flex items-center gap-3">
                      <span className="text-amber-500 font-bold w-4">#{i+1}</span>
                      <div>
                        <div className="font-semibold text-sm">{team?.teamName}</div>
                        <div className="text-[10px] text-white/40 uppercase">{team?.schoolName}</div>
                      </div>
                    </div>
                    <div className="text-xl font-mono text-emerald-400 font-bold">
                      {s.rawValue}{activeEvent?.type.includes('time') ? 's' : ''}
                    </div>
                  </div>
                );
              })}
          </div>
        </div>
      )}
    </div>
  );
}

// --- Admin Event Config ---

function AdminEventsView({ events }) {
  const [showAdd, setShowAdd] = useState(false);
  const [formData, setFormData] = useState({
    name: '',
    type: 'score', 
    method: 'absolute',
    instructions: '',
    leaderBonus: 10,
    tiers: [
      { threshold: 5, points: 95 },
      { threshold: 10, points: 80 }
    ]
  });

  const addEvent = async () => {
    const eventsRef = collection(db, 'artifacts', appId, 'public', 'data', 'events');
    try {
      await addDoc(eventsRef, { ...formData, createdAt: new Date().toISOString() });
      setShowAdd(false);
      setFormData({ 
        name: '', type: 'score', method: 'absolute', instructions: '', 
        leaderBonus: 10, tiers: [{ threshold: 5, points: 95 }, { threshold: 10, points: 80 }] 
      });
    } catch (err) { console.error(err); }
  };

  const deleteEvent = async (id) => {
    try {
      await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'events', id));
    } catch (err) { console.error(err); }
  };

  const addTier = () => {
    setFormData({ ...formData, tiers: [...formData.tiers, { threshold: 0, points: 0 }] });
  };

  const removeTier = (index) => {
    const newTiers = [...formData.tiers];
    newTiers.splice(index, 1);
    setFormData({ ...formData, tiers: newTiers });
  };

  const updateTier = (index, field, value) => {
    const newTiers = [...formData.tiers];
    newTiers[index][field] = parseFloat(value) || 0;
    setFormData({ ...formData, tiers: newTiers });
  };

  return (
    <div className="space-y-6">
      <div className="flex justify-between items-center">
        <h3 className="text-lg font-bold text-slate-800">Event Configuration</h3>
        <button onClick={() => setShowAdd(!showAdd)} className="bg-emerald-600 text-white px-4 py-2 rounded-lg flex gap-2 items-center hover:bg-emerald-700 shadow-sm transition-all">
          {showAdd ? 'Cancel' : <><Plus size={18} /> Add Event</>}
        </button>
      </div>

      {showAdd && (
        <div className="bg-white p-6 rounded-xl border-2 border-emerald-100 shadow-xl space-y-6 animate-in zoom-in-95 duration-200">
          <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
            <div>
              <label className="block text-sm font-bold text-slate-700 mb-1">Event Name</label>
              <input className="w-full p-2 border border-slate-200 rounded-lg" value={formData.name} onChange={e => setFormData({...formData, name: e.target.value})} />
            </div>
            <div>
              <label className="block text-sm font-bold text-slate-700 mb-1">Entry Value Type</label>
              <select className="w-full p-2 border border-slate-200 rounded-lg" value={formData.type} onChange={e => setFormData({...formData, type: e.target.value})}>
                <option value="score">Raw Score (High Wins)</option>
                <option value="time-fast">Time (Fastest Wins)</option>
                <option value="time-long">Time (Longest Wins)</option>
              </select>
            </div>
          </div>
          
          <div className="bg-slate-50 p-4 rounded-xl border border-slate-200">
            <h4 className="text-sm font-bold text-slate-800 mb-4">Scoring Calculation Method</h4>
            <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
              <button 
                className={`p-4 rounded-xl border-2 text-left transition-all ${formData.method === 'absolute' ? 'border-emerald-500 bg-emerald-50' : 'bg-white border-slate-200 opacity-60'}`}
                onClick={() => setFormData({...formData, method: 'absolute'})}
              >
                <div className="font-bold text-slate-900">Absolute Scoring</div>
                <div className="text-[10px] text-slate-500">Judge enters raw points that are used directly.</div>
              </button>
              <button 
                className={`p-4 rounded-xl border-2 text-left transition-all ${formData.method === 'relative' ? 'border-emerald-500 bg-emerald-50' : 'bg-white border-slate-200 opacity-60'}`}
                onClick={() => setFormData({...formData, method: 'relative'})}
              >
                <div className="font-bold text-slate-900">Relative Proximity Brackets</div>
                <div className="text-[10px] text-slate-500">Gap from leader determines point tiers.</div>
              </button>
            </div>
          </div>

          {formData.method === 'relative' && (
            <div className="bg-amber-50 p-6 rounded-xl border border-amber-100 space-y-6 animate-in slide-in-from-top-4">
              <div className="flex justify-between items-center">
                <h4 className="text-sm font-bold text-amber-900 flex items-center gap-2"><Award size={18}/> Proximity Tiers</h4>
                <div className="flex items-center gap-2">
                  <label className="text-[10px] font-bold text-amber-700 uppercase">Winner Bonus:</label>
                  <input type="number" className="w-16 p-1 border border-amber-200 rounded text-center text-sm font-bold" value={formData.leaderBonus} onChange={e => setFormData({...formData, leaderBonus: parseFloat(e.target.value) || 0})} />
                </div>
              </div>

              <div className="space-y-3">
                {formData.tiers.sort((a,b) => a.threshold - b.threshold).map((tier, idx) => (
                  <div key={idx} className="flex items-center gap-4 bg-white p-3 rounded-lg border border-amber-100">
                    <div className="flex-1">
                      <label className="text-[10px] font-bold text-slate-400 block mb-1 uppercase tracking-tighter">Threshold (Within...)</label>
                      <input type="number" className="w-full p-2 border border-slate-100 rounded bg-slate-50 text-sm" value={tier.threshold} onChange={e => updateTier(idx, 'threshold', e.target.value)} />
                    </div>
                    <div className="w-24">
                      <label className="text-[10px] font-bold text-slate-400 block mb-1 uppercase tracking-tighter">Points</label>
                      <input type="number" className="w-full p-2 border border-slate-100 rounded bg-emerald-50 text-emerald-700 font-bold text-sm" value={tier.points} onChange={e => updateTier(idx, 'points', e.target.value)} />
                    </div>
                    <button onClick={() => removeTier(idx)} className="mt-5 p-2 text-slate-300 hover:text-red-500"><Trash2 size={16} /></button>
                  </div>
                ))}
                <button onClick={addTier} className="w-full py-2 border-2 border-dashed border-amber-200 rounded-lg text-amber-600 text-xs font-bold hover:bg-amber-100 transition-colors flex items-center justify-center gap-2"><ListPlus size={14} /> Add Tier</button>
              </div>
            </div>
          )}

          <div>
            <label className="block text-sm font-bold text-slate-700 mb-1">Judge Instructions</label>
            <textarea className="w-full p-2 border border-slate-200 rounded-lg h-24" value={formData.instructions} onChange={e => setFormData({...formData, instructions: e.target.value})} />
          </div>
          <button onClick={addEvent} disabled={!formData.name} className="w-full py-3 bg-emerald-600 text-white font-bold rounded-xl shadow-lg shadow-emerald-50 hover:bg-emerald-700 disabled:opacity-50">Create Competition Event</button>
        </div>
      )}

      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
        {events.map(event => (
          <div key={event.id} className="bg-white p-5 rounded-xl border border-slate-200 flex flex-col justify-between hover:border-emerald-200 transition-all shadow-sm">
            <div>
              <div className="flex justify-between items-start mb-4">
                <h4 className="font-bold text-slate-900">{event.name}</h4>
                <button onClick={() => deleteEvent(event.id)} className="text-slate-300 hover:text-red-500 p-1"><Trash2 size={16} /></button>
              </div>
              <div className="flex gap-2 mb-4">
                <span className="text-[10px] font-bold uppercase px-2 py-1 rounded bg-slate-100 text-slate-500 border border-slate-200">{event.type}</span>
                <span className={`text-[10px] font-bold uppercase px-2 py-1 rounded border ${event.method === 'relative' ? 'bg-amber-50 text-amber-600 border-amber-100' : 'bg-blue-50 text-blue-600 border-blue-100'}`}>{event.method}</span>
              </div>
              {event.method === 'relative' && (
                <div className="mb-4 space-y-1">
                  <div className="text-[9px] font-bold text-slate-400 uppercase">Tiers:</div>
                  <div className="flex flex-wrap gap-1">
                    {event.tiers?.map((t, i) => <span key={i} className="text-[10px] bg-slate-50 border border-slate-100 px-2 py-0.5 rounded text-slate-500">&lt;{t.threshold}: <strong>{t.points}pt</strong></span>)}
                  </div>
                </div>
              )}
            </div>
          </div>
        ))}
      </div>
    </div>
  );
}

// --- Team Management Views ---

function CoachTeamsView({ user, teams }) {
  const [showForm, setShowForm] = useState(false);
  const [editingTeam, setEditingTeam] = useState(null);

  const saveTeam = async (formData) => {
    const teamsRef = collection(db, 'artifacts', appId, 'users', user.uid, 'teams');
    const publicRef = collection(db, 'artifacts', appId, 'public', 'data', 'registrations');
    const teamData = { ...formData, coachName: user.displayName, coachEmail: user.email, updatedAt: new Date().toISOString() };

    try {
      if (editingTeam) {
        await updateDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'teams', editingTeam.id), teamData);
        await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'registrations', editingTeam.id), teamData);
      } else {
        const docRef = await addDoc(teamsRef, teamData);
        await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'registrations', docRef.id), teamData);
      }
      setShowForm(false);
      setEditingTeam(null);
    } catch (err) { console.error(err); }
  };

  const deleteTeam = async (id) => {
    try {
      await deleteDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'teams', id));
      await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'registrations', id));
    } catch (err) { console.error(err); }
  };

  return (
    <div className="space-y-6">
      <div className="flex justify-between items-center">
        <h3 className="text-lg font-bold text-slate-800">Your Registered Teams</h3>
        <button onClick={() => { setEditingTeam(null); setShowForm(true); }} className="bg-blue-600 text-white px-4 py-2 rounded-lg flex gap-2 items-center hover:bg-blue-700 shadow-sm"><Plus size={18} /> Register Team</button>
      </div>
      {showForm && <TeamForm onCancel={() => setShowForm(false)} onSubmit={saveTeam} initialData={editingTeam} />}
      <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
        {teams.map(team => (
          <div key={team.id} className="bg-white p-5 rounded-xl border border-slate-200 flex flex-col justify-between hover:shadow-md transition-all shadow-sm">
            <div>
              <div className="flex justify-between items-start mb-2">
                <h4 className="font-bold text-slate-900 text-lg">{team.teamName}</h4>
                <div className="flex gap-2">
                  <button onClick={() => { setEditingTeam(team); setShowForm(true); }} className="p-1.5 text-slate-400 hover:text-blue-600 rounded"><Edit2 size={16} /></button>
                  <button onClick={() => deleteTeam(team.id)} className="p-1.5 text-slate-400 hover:text-red-600 rounded"><Trash2 size={16} /></button>
                </div>
              </div>
              <p className="text-sm text-slate-500 mb-4">{team.schoolName}</p>
            </div>
          </div>
        ))}
      </div>
    </div>
  );
}

function TeamForm({ onCancel, onSubmit, initialData }) {
  const [formData, setFormData] = useState(initialData || { teamName: '', schoolName: '', students: ['', '', ''] });
  const updateStudent = (val, i) => { const s = [...formData.students]; s[i] = val; setFormData({...formData, students: s}); };
  return (
    <div className="bg-white p-6 rounded-xl border-2 border-blue-100 shadow-xl space-y-4">
      <div className="grid grid-cols-2 gap-4">
        <input className="p-2 border border-slate-200 rounded-lg" placeholder="Team Name" value={formData.teamName} onChange={e => setFormData({...formData, teamName: e.target.value})} />
        <input className="p-2 border border-slate-200 rounded-lg" placeholder="School Name" value={formData.schoolName} onChange={e => setFormData({...formData, schoolName: e.target.value})} />
      </div>
      <div className="space-y-2">
        {formData.students.map((s, i) => <input key={i} className="w-full p-2 border border-slate-200 rounded-lg" placeholder={`Student ${i+1}`} value={s} onChange={e => updateStudent(e.target.value, i)} />)}
      </div>
      <div className="flex justify-end gap-2 pt-4">
        <button onClick={onCancel} className="px-4 py-2 text-slate-500 font-bold">Cancel</button>
        <button onClick={() => onSubmit(formData)} className="px-6 py-2 bg-blue-600 text-white font-bold rounded-lg">Save Team</button>
      </div>
    </div>
  );
}
